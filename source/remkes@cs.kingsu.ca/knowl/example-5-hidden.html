\begin{gather*} (-3, -6. 2) = (0,4,0) = (-3, -10, 2) \\ (-7, 0. 1) = (0,4,0) = (-7, -4, 1) \end{gather*}

\begin{align*} (-3, -10, 2) \times (-7, -4, 1) \amp = (-10 - (-8), (-14, - (-3), 12 - 70))\\ \amp = (-2, -11, -58) \end{align*}

\begin{equation*} (-2, -11, -58) \cdot (0,4,) = 0 - 44 + 0 = -44 \end{equation*}
