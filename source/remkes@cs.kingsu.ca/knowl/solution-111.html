\begin{align*} F(\gamma(t)) \amp = ((t^2-1)^2, 3t - (-4t), (3t)^2 - (t^2-1)(-4t)) \\ \amp = (t^4 - 2t^2 + 1, 7t, 9t^4 - 4t^3 + 4t^2) \end{align*}

\begin{align*} \gamma^\prime(t) \amp = (3, 2t, -4) \end{align*}
